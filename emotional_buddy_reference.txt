import 'package:flutter/material.dart';
import '../widgets/orb_widget.dart';

enum ConversationState {
  idle,
  listening,
  processing,
  thinking,
  speaking,
}

class EmotionalBuddyScreen extends StatelessWidget {
  final ConversationState state;
  final String displayText;
  final bool isActive;
  final VoidCallback onStart;
  final VoidCallback onEnd;

  const EmotionalBuddyScreen({
    super.key,
    required this.state,
    required this.displayText,
    required this.isActive,
    required this.onStart,
    required this.onEnd,
  });

  @override
  Widget build(BuildContext context) {
    return _BuddyView(
      state: state,
      displayText: displayText,
      isActive: isActive,
      onStart: onStart,
      onEnd: onEnd,
    );
  }
}

class _BuddyView extends StatefulWidget {
  final ConversationState state;
  final String displayText;
  final bool isActive;
  final VoidCallback onStart;
  final VoidCallback onEnd;

  const _BuddyView({
    required this.state,
    required this.displayText,
    required this.isActive,
    required this.onStart,
    required this.onEnd,
  });

  @override
  State<_BuddyView> createState() => _BuddyViewState();
}

class _BuddyViewState extends State<_BuddyView>
    with SingleTickerProviderStateMixin {
  late AnimationController _entryCtrl;
  late Animation<double> _entryFade;
  late Animation<Offset> _entrySlide;

  @override
  void initState() {
    super.initState();
    _entryCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 800),
    )..forward();

    _entryFade = CurvedAnimation(
      parent: _entryCtrl,
      curve: Curves.easeOut,
    );

    _entrySlide = Tween<Offset>(
      begin: const Offset(0, 0.08),
      end: Offset.zero,
    ).animate(CurvedAnimation(parent: _entryCtrl, curve: Curves.easeOut));
  }

  @override
  void dispose() {
    _entryCtrl.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: const Color(0xFF050510),
      body: FadeTransition(
        opacity: _entryFade,
        child: SlideTransition(
          position: _entrySlide,
          child: Stack(
            children: [
              _buildBackground(widget.state),
              SafeArea(
                child: Column(
                  children: [
                    _buildTopBar(context),
                    Expanded(
                      child: Column(
                        mainAxisAlignment: MainAxisAlignment.center,
                        children: [
                          _StateLabel(state: widget.state),
                          const SizedBox(height: 32),
                          OrbWidget(state: widget.state, size: 200),
                          const SizedBox(height: 40),
                          _DisplayText(text: widget.displayText),
                        ],
                      ),
                    ),
                    _buildBottomControls(),
                  ],
                ),
              ),
            ],
          ),
        ),
      ),
    );
  }

  Widget _buildBackground(ConversationState state) {
    final color = _bgColorFor(state);
    return AnimatedContainer(
      duration: const Duration(milliseconds: 700),
      decoration: BoxDecoration(
        gradient: RadialGradient(
          center: const Alignment(0, -0.2),
          radius: 1.4,
          colors: [color.withOpacity(0.18), const Color(0xFF050510)],
        ),
      ),
    );
  }

  Color _bgColorFor(ConversationState s) {
    switch (s) {
      case ConversationState.idle:
        return const Color(0xFF6B73FF);
      case ConversationState.listening:
        return const Color(0xFF00D2A0);
      case ConversationState.processing:
        return const Color(0xFFFF6B6B);
      case ConversationState.thinking:
        return const Color(0xFFFF4499);
      case ConversationState.speaking:
        return const Color(0xFF4ECDC4);
    }
  }

  Widget _buildTopBar(BuildContext context) {
    return Padding(
      padding: const EdgeInsets.fromLTRB(16, 8, 16, 0),
      child: Row(
        children: [
          _GlassButton(
            onTap: () => Navigator.pop(context),
            child: const Icon(
              Icons.arrow_back_ios_new_rounded,
              color: Colors.white,
              size: 18,
            ),
          ),
          const Spacer(),
          Column(
            children: [
              const Text(
                'Buddy',
                style: TextStyle(
                  color: Colors.white,
                  fontSize: 17,
                  fontWeight: FontWeight.w600,
                ),
              ),
              Text(
                'Your Emotional Companion',
                style: TextStyle(
                  color: Colors.white.withOpacity(0.5),
                  fontSize: 11,
                ),
              ),
            ],
          ),
          const Spacer(),
          const SizedBox(width: 44),
        ],
      ),
    );
  }

  Widget _buildBottomControls() {
    return Padding(
      padding: const EdgeInsets.fromLTRB(24, 0, 24, 40),
      child: Column(
        children: [
          if (!widget.isActive)
            _StartButton(onTap: widget.onStart)
          else
            _EndButton(onTap: widget.onEnd),
        ],
      ),
    );
  }
}

class _StateLabel extends StatelessWidget {
  final ConversationState state;
  const _StateLabel({required this.state});

  String get _label {
    switch (state) {
      case ConversationState.idle:
        return 'Tap to start';
      case ConversationState.listening:
        return 'Listening...';
      case ConversationState.processing:
        return 'Processing...';
      case ConversationState.thinking:
        return 'Buddy is thinking...';
      case ConversationState.speaking:
        return 'Buddy is speaking...';
    }
  }

  @override
  Widget build(BuildContext context) {
    return AnimatedSwitcher(
      duration: const Duration(milliseconds: 400),
      transitionBuilder: (child, anim) =>
          FadeTransition(opacity: anim, child: child),
      child: Text(
        _label,
        key: ValueKey(_label),
        style: TextStyle(
          color: Colors.white.withOpacity(0.6),
          fontSize: 15,
          fontWeight: FontWeight.w400,
          letterSpacing: 0.5,
        ),
      ),
    );
  }
}

import 'dart:math' as math;
import 'package:flutter/material.dart';

enum ConversationState {
  idle,
  listening,
  processing,
  thinking,
  speaking,
}

class OrbWidget extends StatefulWidget {
  final ConversationState state;
  final double size;

  const OrbWidget({super.key, required this.state, this.size = 260});

  @override
  State<OrbWidget> createState() => _OrbWidgetState();
}

class _OrbWidgetState extends State<OrbWidget> with TickerProviderStateMixin {
  late AnimationController _breatheCtrl;
  late Animation<double> _breatheAnim;
  late AnimationController _rotateCtrl;
  late Animation<double> _rotateAnim;
  late AnimationController _rotate2Ctrl;
  late Animation<double> _rotate2Anim;
  late AnimationController _rippleCtrl;
  late Animation<double> _rippleAnim;
  late Animation<double> _rippleOpacity;
  late AnimationController _thinkCtrl;
  late Animation<double> _thinkAnim;
  late AnimationController _speakCtrl;
  late Animation<double> _speakAnim;
  late AnimationController _colorCtrl;
  late Animation<double> _colorAnim;

  ConversationState _prevState = ConversationState.idle;

  @override
  void initState() {
    super.initState();

    _breatheCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 3000),
    )..repeat(reverse: true);
    _breatheAnim = CurvedAnimation(
      parent: _breatheCtrl,
      curve: Curves.easeInOutSine,
    );

    _rotateCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 8000),
    )..repeat();
    _rotateAnim = Tween<double>(begin: 0, end: 2 * math.pi).animate(_rotateCtrl);

    _rotate2Ctrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 12000),
    )..repeat();
    _rotate2Anim = Tween<double>(begin: 2 * math.pi, end: 0).animate(_rotate2Ctrl);

    _rippleCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 1600),
    )..repeat();
    _rippleAnim = Tween<double>(
      begin: 0.7,
      end: 1.4,
    ).animate(CurvedAnimation(parent: _rippleCtrl, curve: Curves.easeOut));
    _rippleOpacity = Tween<double>(
      begin: 0.6,
      end: 0.0,
    ).animate(CurvedAnimation(parent: _rippleCtrl, curve: Curves.easeOut));

    _thinkCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 1200),
    )..repeat();
    _thinkAnim = Tween<double>(begin: 0, end: 2 * math.pi).animate(_thinkCtrl);

    _speakCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 600),
    )..repeat(reverse: true);
    _speakAnim = CurvedAnimation(
      parent: _speakCtrl,
      curve: Curves.easeInOutSine,
    );

    _colorCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 700),
    );
    _colorAnim = CurvedAnimation(parent: _colorCtrl, curve: Curves.easeInOut);
  }

  @override
  void didUpdateWidget(OrbWidget oldWidget) {
    super.didUpdateWidget(oldWidget);
    if (oldWidget.state != widget.state) {
      _prevState = oldWidget.state;
      _colorCtrl.forward(from: 0);

      switch (widget.state) {
        case ConversationState.idle:
          _breatheCtrl.duration = const Duration(milliseconds: 3500);
          _rotateCtrl.duration = const Duration(milliseconds: 10000);
          break;
        case ConversationState.listening:
          _breatheCtrl.duration = const Duration(milliseconds: 2000);
          _rotateCtrl.duration = const Duration(milliseconds: 6000);
          break;
        case ConversationState.processing:
          _breatheCtrl.duration = const Duration(milliseconds: 1500);
          _rotateCtrl.duration = const Duration(milliseconds: 4000);
          break;
        case ConversationState.thinking:
          _breatheCtrl.duration = const Duration(milliseconds: 1000);
          _rotateCtrl.duration = const Duration(milliseconds: 2500);
          break;
        case ConversationState.speaking:
          _breatheCtrl.duration = const Duration(milliseconds: 800);
          _rotateCtrl.duration = const Duration(milliseconds: 3500);
          break;
      }
    }
  }

  @override
  void dispose() {
    _breatheCtrl.dispose();
    _rotateCtrl.dispose();
    _rotate2Ctrl.dispose();
    _rippleCtrl.dispose();
    _thinkCtrl.dispose();
    _speakCtrl.dispose();
    _colorCtrl.dispose();
    super.dispose();
  }

  List<Color> _colorsFor(ConversationState s) {
    switch (s) {
      case ConversationState.idle:
        return [
          const Color(0xFF6B73FF),
          const Color(0xFF9B6DFF),
          const Color(0xFF000DFF),
        ];
      case ConversationState.listening:
        return [
          const Color(0xFF00D2A0),
          const Color(0xFF00A8CC),
          const Color(0xFF0066FF),
        ];
      case ConversationState.processing:
        return [
          const Color(0xFFFFB347),
          const Color(0xFFFF6B6B),
          const Color(0xFFFF4499),
        ];
      case ConversationState.thinking:
        return [
          const Color(0xFFFF6B6B),
          const Color(0xFFFF4499),
          const Color(0xFF9B6DFF),
        ];
      case ConversationState.speaking:
        return [
          const Color(0xFF4ECDC4),
          const Color(0xFF44A8B3),
          const Color(0xFF6B73FF),
        ];
    }
  }

  List<Color> _lerpColorList(List<Color> a, List<Color> b, double t) {
    return List.generate(a.length, (i) => Color.lerp(a[i], b[i], t)!);
  }

  @override
  Widget build(BuildContext context) {
    return AnimatedBuilder(
      animation: Listenable.merge([
        _breatheAnim,
        _rotateAnim,
        _rotate2Anim,
        _rippleAnim,
        _thinkAnim,
        _speakAnim,
        _colorAnim,
      ]),
      builder: (context, _) {
        final t = _colorAnim.value;
        final prevColors = _colorsFor(_prevState);
        final currColors = _colorsFor(widget.state);
        final colors = _lerpColorList(prevColors, currColors, t);

        final isListening = widget.state == ConversationState.listening;
        final isSpeaking = widget.state == ConversationState.speaking;
        final isThinking =
            widget.state == ConversationState.thinking ||
            widget.state == ConversationState.processing;

        double coreScale = 1.0 + _breatheAnim.value * 0.06;
        if (isSpeaking) {
          coreScale += _speakAnim.value * 0.10;
        }
        if (isThinking) {
          coreScale += math.sin(_thinkAnim.value * 3) * 0.04;
        }

        final sz = widget.size;

        return SizedBox(
          width: sz * 1.5,
          height: sz * 1.5,
          child: Stack(
            alignment: Alignment.center,
            children: [
              if (isListening) ...[
                _buildRipple(
                  sz,
                  colors[0].withOpacity(_rippleOpacity.value),
                  _rippleAnim.value,
                ),
                _buildRipple(
                  sz,
                  colors[1].withOpacity(_rippleOpacity.value * 0.5),
                  _rippleAnim.value * 1.15,
                ),
              ],
              Transform.rotate(
                angle: _rotateAnim.value,
                child: Container(
                  width: sz * 1.18,
                  height: sz * 1.18,
                  decoration: BoxDecoration(
                    shape: BoxShape.circle,
                    gradient: SweepGradient(
                      colors: [
                        colors[0].withOpacity(0.0),
                        colors[0].withOpacity(0.45),
                        colors[1].withOpacity(0.35),
                        colors[2].withOpacity(0.20),
                        colors[0].withOpacity(0.0),
                      ],
                    ),
                  ),
                ),
              ),
              Transform.rotate(
                angle: _rotate2Anim.value,
                child: Container(
                  width: sz * 1.06,
                  height: sz * 1.06,
                  decoration: BoxDecoration(
                    shape: BoxShape.circle,
                    gradient: SweepGradient(
                      colors: [
                        colors[2].withOpacity(0.0),
                        colors[2].withOpacity(0.30),
                        colors[1].withOpacity(0.25),
                        colors[2].withOpacity(0.0),
                      ],
                    ),
                  ),
                ),
              ),
              if (isThinking) _buildOrbitDots(sz, colors, _thinkAnim.value),
              Transform.scale(
                scale: coreScale,
                child: Container(
                  width: sz,
                  height: sz,
                  decoration: BoxDecoration(
                    shape: BoxShape.circle,
                    gradient: RadialGradient(
                      colors: [
                        colors[0].withOpacity(0.9),
                        colors[1].withOpacity(0.75),
                        colors[2].withOpacity(0.6),
                        Colors.black.withOpacity(0.4),
                      ],
                      stops: const [0.0, 0.4, 0.75, 1.0],
                      center: const Alignment(-0.3, -0.3),
                    ),
                    boxShadow: [
                      BoxShadow(
                        color: colors[0].withOpacity(0.5),
                        blurRadius: sz * 0.35,
                        spreadRadius: sz * 0.05,
                      ),
                      BoxShadow(
                        color: colors[1].withOpacity(0.3),
                        blurRadius: sz * 0.55,
                        spreadRadius: sz * 0.02,
                      ),
                    ],
                  ),
                ),
              ),
              Transform.scale(
                scale: coreScale,
                child: Container(
                  width: sz * 0.38,
                  height: sz * 0.22,
                  decoration: BoxDecoration(
                    borderRadius: BorderRadius.circular(sz * 0.2),
                    gradient: RadialGradient(
                      colors: [
                        Colors.white.withOpacity(0.55),
                        Colors.white.withOpacity(0.0),
                      ],
                    ),
                  ),
                ),
              ),
              if (isSpeaking) _buildWaveform(sz, colors, _speakAnim.value),
            ],
          ),
        );
      },
    );
  }

  Widget _buildRipple(double sz, Color color, double scale) {
    return Container(
      width: sz * scale,
      height: sz * scale,
      decoration: BoxDecoration(
        shape: BoxShape.circle,
        border: Border.all(color: color, width: 2),
      ),
    );
  }

  Widget _buildOrbitDots(double sz, List<Color> colors, double angle) {
    const dotCount = 5;
    return SizedBox(
      width: sz * 1.3,
      height: sz * 1.3,
      child: Stack(
        alignment: Alignment.center,
        children: List.generate(dotCount, (i) {
          final dotAngle = angle + (2 * math.pi * i / dotCount);
          final radius = sz * 0.52;
          final x = math.cos(dotAngle) * radius;
          final y = math.sin(dotAngle) * radius;
          final dotSize = sz * (0.048 + 0.016 * math.sin(dotAngle * 2));
          return Transform.translate(
            offset: Offset(x, y),
            child: Container(
              width: dotSize,
              height: dotSize,
              decoration: BoxDecoration(
                shape: BoxShape.circle,
                color: colors[i % colors.length].withOpacity(0.85),
                boxShadow: [
                  BoxShadow(
                    color: colors[i % colors.length].withOpacity(0.5),
                    blurRadius: dotSize * 1.5,
                  ),
                ],
              ),
            ),
          );
        }),
      ),
    );
  }

  Widget _buildWaveform(double sz, List<Color> colors, double pulse) {
    const barCount = 7;
    final heights = List.generate(barCount, (i) {
      final phase = i / barCount * math.pi * 2;
      return 0.3 + 0.7 * math.pow(math.sin(phase + pulse * math.pi), 2).abs();
    });

    return SizedBox(
      width: sz * 0.55,
      height: sz * 0.35,
      child: Row(
        mainAxisAlignment: MainAxisAlignment.spaceEvenly,
        crossAxisAlignment: CrossAxisAlignment.center,
        children: List.generate(barCount, (i) {
          return AnimatedContainer(
            duration: const Duration(milliseconds: 80),
            width: sz * 0.035,
            height: sz * 0.25 * heights[i],
            decoration: BoxDecoration(
              borderRadius: BorderRadius.circular(sz * 0.02),
              color: Colors.white.withOpacity(0.75),
              boxShadow: [
                BoxShadow(color: colors[0].withOpacity(0.4), blurRadius: 4),
              ],
            ),
          );
        }),
      ),
    );
  }
}

4. HOW THE UI EFFECTS WORK
────────────────────────────────────────────────────────────────────────────────

1. Screen entry animation
   - The screen fades in using FadeTransition.
   - The screen also slides upward slightly using SlideTransition.

2. Background color transition
   - The background uses AnimatedContainer.
   - A radial gradient shifts to a different accent color when the state changes.
   - The transition duration is 700ms.

3. Orb breathing effect
   - The orb always has a slow breathing scale animation.
   - This keeps the orb alive even when idle.

4. Listening effect
   - Expanding ripple rings appear around the orb.
   - The rings fade out as they grow.

5. Thinking effect
   - Small glowing orbit dots rotate around the orb.
   - The orb pulse and ring rotation both become faster.

6. Speaking effect
   - Waveform bars appear in the center of the orb.
   - The orb expands and contracts more actively.

7. Halo ring depth
   - Two sweep-gradient circular rings rotate in opposite directions.
   - This adds motion depth and a more premium visual feel.

8. Gloss highlight

────────────────────────────────────────────────────────────────────────────────
6. HOW THE COLOR BLENDING HAPPENS
────────────────────────────────────────────────────────────────────────────────

The orb stores the previous state in _prevState.

When the state changes:
1. _colorCtrl.forward(from: 0) starts.
2. _colorAnim runs from 0 to 1 over 700ms.
3. _lerpColorList() blends the old palette into the new palette.
4. All orb layers use the blended colors during that time.

This prevents harsh jumps and creates smooth transitions such as:
- listening -> thinking
- thinking -> speaking
- speaking -> listening

import 'dart:async';
import 'dart:convert';
import 'package:flutter/foundation.dart';
import 'package:shared_preferences/shared_preferences.dart';
import '../models/conversation_message.dart';
import '../services/audio_service.dart';
import '../services/groq_service.dart';
import '../services/sarvam_tts_service.dart';

enum ConversationState {
  idle,
  listening,
  processing, // transcription in progress
  thinking,   // LLM generating
  speaking,   // TTS playing
}

class ConversationProvider extends ChangeNotifier {
  final _audio = AudioService();
  final _groq  = GroqService();
  final _tts   = SarvamTtsService();

  // ── State ─────────────────────────────────────────────────────────
  ConversationState _state = ConversationState.idle;
  ConversationState get state => _state;

  bool _active = false;
  bool get isActive => _active;

  // ── Current turn display text ─────────────────────────────────────
  String _displayText = '';
  String get displayText => _displayText;

  // ── Session messages ─────────────────────────────────────────────
  final List<ConversationMessage> _messages = [];
  List<ConversationMessage> get messages => List.unmodifiable(_messages);

  List<ConversationMessage> _lastSessionMessages = [];
  List<ConversationMessage> get lastSessionMessages =>
      List.unmodifiable(_lastSessionMessages);

  // ignore: unused_field
  String? _lastRecordingPath;

  // ── Streaming TTS pipeline state ─────────────────────────────────
  final List<String> _ttsPhraseQueue = [];
  bool _ttsProcessing = false;
  bool _llmStreaming   = false;

  // ── Init ─────────────────────────────────────────────────────────
  ConversationProvider() {
    _audio.onSilenceDetected = _onSilenceDetected;
    _loadLastSession();
  }

  // ── Start / End conversation ──────────────────────────────────────
  Future<void> startConversation() async {
    if (_active) return;

    final hasPermission = await _audio.requestMicPermission();
    if (!hasPermission) {
      _displayText = 'Microphone permission is required.';
      notifyListeners();
      return;
    }

    _active = true;
    _messages.clear();
    _ttsPhraseQueue.clear();
    _ttsProcessing = false;
    _llmStreaming   = false;
    _displayText    = "I'm listening…";
    _setState(ConversationState.listening);

    await _aiOpensConversation();
  }

  Future<void> endConversation() async {
    if (!_active) return;
    _active      = false;
    _displayText = '';
    _setState(ConversationState.idle);

    await _audio.stopRecording();
    await _audio.stopPlayback();
    await _tts.cleanupTempFiles();
    await _saveSession();
    notifyListeners();
  }

  // ── AI opens the session with a warm greeting ─────────────────────
  Future<void> _aiOpensConversation() async {
    _setState(ConversationState.thinking);
    _displayText = 'Buddy is here…';
    notifyListeners();

    _messages.add(
      ConversationMessage(
        id: _id(),
        role: MessageRole.user,
        content: '[Session started]',
        timestamp: DateTime.now(),
      ),
    );

    await _runLlmAndSpeak();
  }

  // ── Called by VAD when the user stops speaking ────────────────────
  Future<void> _onSilenceDetected() async {
    if (!_active) return;
    _setState(ConversationState.processing);
    _displayText = 'Processing…';
    notifyListeners();

    final path = await _audio.stopRecording();
    if (path == null) {
      if (_active) await _startListening();
      return;
    }

    try {
      final transcript = await _groq.transcribeAudio(path);
      if (transcript.isEmpty) {
        if (_active) await _startListening();
        return;
      }

      _messages.add(
        ConversationMessage(
          id: _id(),
          role: MessageRole.user,
          content: transcript,
          timestamp: DateTime.now(),
        ),
      );
      _displayText = 'You: $transcript';
      notifyListeners();

      await _runLlmAndSpeak();
    } catch (e) {
      if (_active) {
        _displayText = 'Something went wrong. Listening again…';
        notifyListeners();
        await _startListening();
      }
    }
  }

  // ── LLM → phrase queue → pre-fetch TTS → gapless playback ────────
  Future<void> _runLlmAndSpeak() async {
    if (!_active) return;
    _setState(ConversationState.thinking);
    _ttsPhraseQueue.clear();
    _ttsProcessing = false;
    _llmStreaming   = true;

    String fullResponse = '';

    try {
      fullResponse = await _groq.streamChatResponse(
        history: _messages,
        onSentence: (phrase) {
          if (!_active) return;
          _ttsPhraseQueue.add(phrase);
          _processTtsQueue();
        },
        onFullResponse: (full) {
          fullResponse = full;
        },
      );
    } catch (e) {
      debugPrint('LLM error: $e');
      _llmStreaming = false;
      if (_active) {
        _ttsPhraseQueue.clear();
        while (_ttsProcessing) {
          await Future.delayed(const Duration(milliseconds: 50));
        }
        _displayText = 'Connection issue. Listening again…';
        notifyListeners();
        await _startListening();
      }
      return;
    } finally {
      _llmStreaming = false;
    }

    while ((_ttsProcessing || _ttsPhraseQueue.isNotEmpty) && _active) {
      await Future.delayed(const Duration(milliseconds: 50));
    }

    if (fullResponse.isNotEmpty) {
      _messages.add(
        ConversationMessage(
          id: _id(),
          role: MessageRole.assistant,
          content: fullResponse,
          timestamp: DateTime.now(),
        ),
      );
      _displayText = fullResponse;
      notifyListeners();
    }

    if (_active) await _startListening();
  }

  // ── Streaming TTS pipeline: 2-slot pre-fetch for gapless playback ─
  Future<void> _processTtsQueue() async {
    if (_ttsProcessing || !_active) return;
    _ttsProcessing = true;

    Future<Uint8List>? prefetchFuture;

    loop:
    while (_active) {
      Future<Uint8List> currentFuture;

      if (prefetchFuture != null) {
        currentFuture  = prefetchFuture;
        prefetchFuture = null;
      } else {
        while (_ttsPhraseQueue.isEmpty) {
          if (!_active) break loop;
          if (!_llmStreaming) break loop;
          await Future.delayed(const Duration(milliseconds: 20));
        }
        if (!_active || _ttsPhraseQueue.isEmpty) break;
        final phrase  = _ttsPhraseQueue.removeAt(0);
        currentFuture = _tts.synthesizeToBytes(phrase);
      }

      if (_ttsPhraseQueue.isNotEmpty) {
        final nextPhrase = _ttsPhraseQueue.removeAt(0);
        prefetchFuture   = _tts.synthesizeToBytes(nextPhrase);
      }

      try {
        final bytes = await currentFuture;
        _setState(ConversationState.speaking);
        if (_active) await _audio.playBytes(bytes);
      } catch (e) {
        debugPrint('TTS/playback error: $e');
      }

      if (_ttsPhraseQueue.isEmpty && prefetchFuture == null && !_llmStreaming) {
        break;
      }
    }

    _ttsProcessing = false;
  }

  // ── Start microphone listening ────────────────────────────────────
  Future<void> _startListening() async {
    if (!_active) return;
    _displayText = "I'm listening…";
    _setState(ConversationState.listening);
    notifyListeners();
    _lastRecordingPath = await _audio.startRecording();
  }

  // ── Persistence (SharedPreferences) ──────────────────────────────
  Future<void> _saveSession() async {
    if (_messages.isEmpty) return;
    try {
      final prefs = await SharedPreferences.getInstance();
      final sessionJson = jsonEncode(
        _messages
            .where((m) => m.content != '[Session started]')
            .map((m) => m.toJson())
            .toList(),
      );
      await prefs.setString('latest_session', sessionJson);
      _lastSessionMessages = _messages
          .where((m) => m.content != '[Session started]')
          .toList();
      notifyListeners();
    } catch (_) {}
  }

  Future<void> _loadLastSession() async {
    try {
      final prefs = await SharedPreferences.getInstance();
      final raw   = prefs.getString('latest_session');
      if (raw == null) return;
      final list  = jsonDecode(raw) as List<dynamic>;
      _lastSessionMessages = list
          .map((j) => ConversationMessage.fromJson(j as Map<String, dynamic>))
          .toList();
      notifyListeners();
    } catch (_) {}
  }

  // ── Helpers ───────────────────────────────────────────────────────
  void _setState(ConversationState s) {
    _state = s;
    notifyListeners();
  }

  String _id() => DateTime.now().microsecondsSinceEpoch.toString();

  @override
  Future<void> dispose() async {
    await _audio.dispose();
    super.dispose();
  }
}

class OrbWidget extends StatefulWidget {

class _OrbWidgetState extends State<OrbWidget> with TickerProviderStateMixin {
  // Core pulse / breathe
  late AnimationController _breatheCtrl;
  late Animation<double>   _breatheAnim;

  // Rotation for the shimmer rings
  late AnimationController _rotateCtrl;
  late Animation<double>   _rotateAnim;

  // Secondary rotation (counter-clockwise)
  late AnimationController _rotate2Ctrl;
  late Animation<double>   _rotate2Anim;

  // Ripple (listening)
  late AnimationController _rippleCtrl;
  late Animation<double>   _rippleAnim;
  late Animation<double>   _rippleOpacity;

  // Thinking spin
  late AnimationController _thinkCtrl;
  late Animation<double>   _thinkAnim;

  // Speaking waveform pulse
  late AnimationController _speakCtrl;
  late Animation<double>   _speakAnim;

  // Colour lerp
  late AnimationController _colorCtrl;
  late Animation<double>   _colorAnim;

  ConversationState _prevState = ConversationState.idle;

  @override
  void initState() {
    super.initState();

    _breatheCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 3000),
    )..repeat(reverse: true);
    _breatheAnim = CurvedAnimation(
      parent: _breatheCtrl,
      curve: Curves.easeInOutSine,
    );

    _rotateCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 8000),
    )..repeat();
    _rotateAnim = Tween<double>(
      begin: 0,
      end: 2 * math.pi,
    ).animate(_rotateCtrl);

    _rotate2Ctrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 12000),
    )..repeat();
    _rotate2Anim = Tween<double>(
      begin: 2 * math.pi,
      end: 0,
    ).animate(_rotate2Ctrl);

    _rippleCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 1600),
    )..repeat();
    _rippleAnim = Tween<double>(
      begin: 0.7,
      end: 1.4,
    ).animate(CurvedAnimation(parent: _rippleCtrl, curve: Curves.easeOut));
    _rippleOpacity = Tween<double>(
      begin: 0.6,
      end: 0.0,
    ).animate(CurvedAnimation(parent: _rippleCtrl, curve: Curves.easeOut));

    _thinkCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 1200),
    )..repeat();
    _thinkAnim = Tween<double>(begin: 0, end: 2 * math.pi).animate(_thinkCtrl);

    _speakCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 600),
    )..repeat(reverse: true);
    _speakAnim = CurvedAnimation(
      parent: _speakCtrl,
      curve: Curves.easeInOutSine,
    );

    _colorCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 700),
    );
    _colorAnim = CurvedAnimation(parent: _colorCtrl, curve: Curves.easeInOut);
  }

  @override
  void didUpdateWidget(OrbWidget oldWidget) {
    super.didUpdateWidget(oldWidget);
    if (oldWidget.state != widget.state) {
      _prevState = oldWidget.state;
      _colorCtrl.forward(from: 0);

      switch (widget.state) {
        case ConversationState.idle:
          _breatheCtrl.duration = const Duration(milliseconds: 3500);
          _rotateCtrl.duration  = const Duration(milliseconds: 10000);
          break;
        case ConversationState.listening:
          _breatheCtrl.duration = const Duration(milliseconds: 2000);
          _rotateCtrl.duration  = const Duration(milliseconds: 6000);
          break;
        case ConversationState.processing:
          _breatheCtrl.duration = const Duration(milliseconds: 1500);
          _rotateCtrl.duration  = const Duration(milliseconds: 4000);
          break;
        case ConversationState.thinking:
          _breatheCtrl.duration = const Duration(milliseconds: 1000);
          _rotateCtrl.duration  = const Duration(milliseconds: 2500);
          break;
        case ConversationState.speaking:
          _breatheCtrl.duration = const Duration(milliseconds: 800);
          _rotateCtrl.duration  = const Duration(milliseconds: 3500);
          break;
      }
    }
  }

  @override

    _breatheCtrl.dispose();
    _rotateCtrl.dispose();
    _rotate2Ctrl.dispose();
    _rippleCtrl.dispose();
    _thinkCtrl.dispose();
    _speakCtrl.dispose();
    _colorCtrl.dispose();
    super.dispose();
  }

  Widget _buildRipple(double sz, Color color, double scale) {
    return Container(
      width: sz * scale,
      height: sz * scale,
      decoration: BoxDecoration(
        shape: BoxShape.circle,
        border: Border.all(color: color, width: 2),
      ),
    );
  }

  Widget _buildOrbitDots(double sz, List<Color> colors, double angle) {
    const dotCount = 5;
    return SizedBox(
      width: sz * 1.3,
      height: sz * 1.3,
      child: Stack(
        alignment: Alignment.center,
        children: List.generate(dotCount, (i) {
          final dotAngle = angle + (2 * math.pi * i / dotCount);
          final radius   = sz * 0.52;
          final x        = math.cos(dotAngle) * radius;
          final y        = math.sin(dotAngle) * radius;
          final dotSize  = sz * (0.048 + 0.016 * math.sin(dotAngle * 2));
          return Transform.translate(
            offset: Offset(x, y),
            child: Container(
              width: dotSize,
              height: dotSize,
              decoration: BoxDecoration(
                shape: BoxShape.circle,
                color: colors[i % colors.length].withOpacity(0.85),
                boxShadow: [
                  BoxShadow(
                    color: colors[i % colors.length].withOpacity(0.5),
                    blurRadius: dotSize * 1.5,
                  ),
                ],
              ),
            ),
          );
        }),
      ),
    );
  }

  Widget _buildWaveform(double sz, List<Color> colors, double pulse) {
    const barCount = 7;
    final heights = List.generate(barCount, (i) {
      final phase = i / barCount * math.pi * 2;
      return 0.3 + 0.7 * math.pow(math.sin(phase + pulse * math.pi), 2).abs();
    });

    return SizedBox(
      width: sz * 0.55,
      height: sz * 0.35,
      child: Row(
        mainAxisAlignment: MainAxisAlignment.spaceEvenly,
        crossAxisAlignment: CrossAxisAlignment.center,
        children: List.generate(barCount, (i) {
          return AnimatedContainer(
            duration: const Duration(milliseconds: 80),
            width: sz * 0.035,
            height: sz * 0.25 * heights[i],
            decoration: BoxDecoration(
              borderRadius: BorderRadius.circular(sz * 0.02),
              color: Colors.white.withOpacity(0.75),
              boxShadow: [
                BoxShadow(color: colors[0].withOpacity(0.4), blurRadius: 4),
              ],
            ),
          );
        }),
      ),
    );
  }
}

    )..forward();
    _entryFade  = CurvedAnimation(parent: _entryCtrl, curve: Curves.easeOut);
    _entrySlide = Tween<Offset>(
      begin: const Offset(0, 0.08),
      end: Offset.zero,
    ).animate(CurvedAnimation(parent: _entryCtrl, curve: Curves.easeOut));
  }

  @override
  void dispose() {
    _entryCtrl.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: const Color(0xFF050510),
      body: Consumer<ConversationProvider>(
        builder: (context, provider, _) {
          return FadeTransition(
            opacity: _entryFade,
            child: SlideTransition(
              position: _entrySlide,
              child: Stack(
                children: [
                  // ── Background gradient ──────────────────────────
                  _buildBackground(provider.state),

                  // ── Main content ─────────────────────────────────
                  SafeArea(
                    child: Column(
                      children: [
                        // Top bar
                        _buildTopBar(context, provider),

                        // Orb + status area
                        Expanded(
                          child: Column(
                            mainAxisAlignment: MainAxisAlignment.center,
                            children: [
                              // State label
                              _StateLabel(state: provider.state),
                              const SizedBox(height: 32),

                              // The animated orb
                              OrbWidget(state: provider.state, size: 200),

                              const SizedBox(height: 40),

                              // Display text (transcript / response)
                              _DisplayText(text: provider.displayText),
                            ],
                          ),
                        ),

                        // Bottom button area
                        _buildBottomControls(context, provider),
                      ],
                    ),
                  ),
                ],
              ),
            ),
          );
        },
      ),
    );
  }

  Widget _buildBackground(ConversationState state) {
    final color = _bgColorFor(state);
    return AnimatedContainer(
      duration: const Duration(milliseconds: 700),
      decoration: BoxDecoration(
        gradient: RadialGradient(
          center: const Alignment(0, -0.2),
          radius: 1.4,
          colors: [color.withOpacity(0.18), const Color(0xFF050510)],
        ),
      ),
    );
  }

  Color _bgColorFor(ConversationState s) {
    switch (s) {
      case ConversationState.idle:
        return const Color(0xFF6B73FF);
      case ConversationState.listening:
        return const Color(0xFF00D2A0);
      case ConversationState.processing:
        return const Color(0xFFFF6B6B);
      case ConversationState.thinking:
        return const Color(0xFFFF4499);
      case ConversationState.speaking:
        return const Color(0xFF4ECDC4);
    }
  }

  Widget _buildTopBar(BuildContext context, ConversationProvider provider) {
    return Padding(
      padding: const EdgeInsets.fromLTRB(16, 8, 16, 0),
      child: Row(
        children: [
          // Back button
          _GlassButton(
            onTap: () async {
              if (provider.isActive) {
                await provider.endConversation();
              }
              if (context.mounted) Navigator.pop(context);
            },
            child: const Icon(
              Icons.arrow_back_ios_new_rounded,
              color: Colors.white,
              size: 18,
            ),
          ),
          const Spacer(),
          // Title
          Column(
            children: [
              const Text(
                'Buddy',
                style: TextStyle(
                  color: Colors.white,
                  fontSize: 17,
                  fontWeight: FontWeight.w600,
                ),
              ),
              Text(
                'Your Emotional Companion',
                style: TextStyle(
                  color: Colors.white.withOpacity(0.5),
                  fontSize: 11,
                ),
              ),
            ],
          ),
          const Spacer(),
          // Spacer to balance the row
          const SizedBox(width: 44),
        ],
      ),
    );
  }

  Widget _buildBottomControls(
    BuildContext context,
    ConversationProvider provider,
  ) {
    return Padding(
      padding: const EdgeInsets.fromLTRB(24, 0, 24, 40),
      child: Column(
        children: [
          if (!provider.isActive)
            _StartButton(onTap: () => provider.startConversation())
          else
            _EndButton(
              onTap: () async {
                await provider.endConversation();
              },
            ),
        ],
      ),
    );
  }
}

// State label – animated crossfade between states
// ─────────────────────────────────────────────────────────────────────────────
class _StateLabel extends StatelessWidget {
  final ConversationState state;
  const _StateLabel({required this.state});

  String get _label {
    switch (state) {
      case ConversationState.idle:
        return 'Tap to start';
      case ConversationState.listening:
        return 'Listening…';
      case ConversationState.processing:
        return 'Processing…';
      case ConversationState.thinking:
        return 'Buddy is thinking…';
      case ConversationState.speaking:
        return 'Buddy is speaking…';
    }
  }

  @override
  Widget build(BuildContext context) {
    return AnimatedSwitcher(
      duration: const Duration(milliseconds: 400),
      transitionBuilder: (child, anim) =>
          FadeTransition(opacity: anim, child: child),
      child: Text(
        _label,
        key: ValueKey(_label),
        style: TextStyle(
          color: Colors.white.withOpacity(0.6),
          fontSize: 15,
          fontWeight: FontWeight.w400,
          letterSpacing: 0.5,
        ),
      ),
    );
  }
}

// ─────────────────────────────────────────────────────────────────────────────
// Start button – large pulsing CTA
// ─────────────────────────────────────────────────────────────────────────────
class _StartButton extends StatefulWidget {
  final VoidCallback onTap;
  const _StartButton({required this.onTap});

  @override
  State<_StartButton> createState() => _StartButtonState();
}

class _StartButtonState extends State<_StartButton>
    with SingleTickerProviderStateMixin {
  late AnimationController _ctrl;
  late Animation<double>   _scale;

  @override
  void initState() {
    super.initState();
    _ctrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 1400),
    )..repeat(reverse: true);
    _scale = Tween<double>(
      begin: 1.0,
      end: 1.06,
    ).animate(CurvedAnimation(parent: _ctrl, curve: Curves.easeInOutSine));
  }

  @override
  void dispose() {
    _ctrl.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return AnimatedBuilder(
      animation: _scale,
      builder: (_, child) =>
          Transform.scale(scale: _scale.value, child: child),
      child: GestureDetector(
        onTap: widget.onTap,
        child: Container(
          width: double.infinity,
          height: 64,
          decoration: BoxDecoration(
            gradient: const LinearGradient(
              colors: [Color(0xFF6B73FF), Color(0xFF9B6DFF)],
            ),
            borderRadius: BorderRadius.circular(32),
            boxShadow: [
              BoxShadow(
                color: const Color(0xFF6B73FF).withOpacity(0.55),
                blurRadius: 24,
                offset: const Offset(0, 8),
              ),
            ],
          ),
          child: const Row(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              Icon(Icons.mic_rounded, color: Colors.white, size: 24),
              SizedBox(width: 12),
              Text(
                'Start Conversation',
                style: TextStyle(
                  color: Colors.white,
                  fontSize: 17,
                  fontWeight: FontWeight.w600,
                  letterSpacing: 0.3,
                ),
              ),
            ],
          ),
        ),
      ),
    );
  }
}

================================================================================
  QUICK SETUP GUIDE FOR ANOTHER ANDROID PROJECT
================================================================================

1. Add the dependencies listed in Section 1 to your pubspec.yaml, then run:
     flutter pub get

2. Copy the Dart files into your project with this folder structure:
     lib/
       models/
         conversation_message.dart    ← Section 2
       providers/
         conversation_provider.dart   ← Section 3  (ConversationState lives here)
       widgets/
         orb_widget.dart              ← Section 4
       screens/
         emotional_buddy_screen.dart  ← Section 5

3. Wire up ConversationProvider in main.dart:
     void main() {
       runApp(
         ChangeNotifierProvider(
           create: (_) => ConversationProvider(),
           child: const MyApp(),
         ),
       );
     }

4. Navigate to the screen from anywhere:
     Navigator.push(
       context,
       MaterialPageRoute(builder: (_) => const EmotionalBuddyScreen()),
     );

5. Android permissions — add to android/app/src/main/AndroidManifest.xml:
     <uses-permission android:name="android.permission.RECORD_AUDIO"/>
     <uses-permission android:name="android.permission.INTERNET"/>

6. The services (AudioService, GroqService, SarvamTtsService) are the backend
   layer. The UI files (screen + orb) have NO dependency on any specific
   service implementation — swap them for your own services by matching the
   same method signatures called by ConversationProvider.
